                      // 1. Функция sum принимает параметром целые положительные
                      // числа (неопределённое кол-во) и возвращает их сумму (rest).

                      export function sum(...nums: Array): number {
                        return nums.reduce((acc, currentValue) => acc += currentValue)
                      }


                      // 2. Функция getTriangleType принимает три параметра:
                      // длины сторон треугольника.
                      // Функция должна возвращать:
                      //  - "10", если треугольник равносторонний,
                      //  - "01", если треугольник равнобедренный,
                      //  - "11", если треугольник обычный,
                      //  - "00", если такого треугольника не существует.

                      export function getTriangleType(a: number, b: number, c: number): string {
                        // Существует ли треугольник
                        if (a + b > c && b + c > a && a + c > b) {
                          // Равносторонний
                          if (a === b && b === c) return '10';
                          // Равнобедренный
                          if (a === b || b === c || a === c) return '01';
                          // Обычный
                          return '11';
                        }
                        // Такого треугольника не существует
                        return '00';
                      }


                      // 3. Функция getSum принимает параметром целое число и возвращает
                      // сумму цифр этого числа

                      export function getSum(number: number): number {
                        return Array.from(number.toString())
                        .reduce((acc, currentValue) => acc += Number(currentValue), 0);
                      }


                      // 4. Функция принимает isEvenIndexSumGreater параметром массив чисел.
                      // Если сумма чисел с чётными ИНДЕКСАМИ!!! (0 как чётный индекс) больше
                      // суммы чисел с нечётными ИНДЕКСАМИ!!!, то функция возвращает true.
                      // В противном случае - false.

                      export const isEvenIndexSumGreater = (arr: Array): boolean => {
                        const even = arr
                        .filter((n, index) => index % 2 === 0)
                        .reduce((acc, currentValue) => acc += currentValue, 0);
                        const odd = arr
                        .filter((n, index) => index % 2 !== 0)
                        .reduce((acc, currentValue) => acc += currentValue, 0);
                        return even > odd
                      }


                      // 5. Функция isSquareGreater принимает два параметра: площадь круга и
                      // площадь квадрата. Функция должна возвращать true если круг не будет выступать за пределы
                      // квадрата и false в противном случае. Центры фигур совпадают.

                      export function isSquareGreater(areaCr: number, areaSq: number): boolean {
                        return areaCr / areaSq <= Math.PI / 4;
                      }


                      // 6. Функция-банкомат принимает параметром целое натуральное число (сумму).
                      // Возвращает массив с наименьшим количеством купюр, которыми можно выдать эту
                      // сумму. Доступны банкноты следующих номиналов:
                      // const banknotes = [1000, 500, 100, 50, 20, 10, 5, 2, 1].
                      // Считаем, что количество банкнот каждого номинала не ограничено

                      // Д.З.:
                      export function getBanknoteList(amountOfMoney: number): Array {
                        const banknotes = [1000, 500, 100, 50, 20, 10, 5, 2, 1]
                        const result: Array = [];
                          let restOfMoney = amountOfMoney;
                          while (restOfMoney > 0) {
                            let banknote = banknotes.find(b => (restOfMoney - b) >= 0);
                            if (banknote) {
                              restOfMoney -= banknote;
                              result.push(banknote);
                            }
                          }
                          return result;
                        }
